5.10.1. еёёз”ЁВ¶

е®үиЈ…:

$ pip install -U scikit-learn
$ conda install scikit-learn

SKlearnеҢ…еҗ«зҡ„жңәеҷЁеӯҰд№ ж–№ејҸ:

еҲҶзұ»пјҢеӣһеҪ’пјҢж— зӣ‘зқЈпјҢж•°жҚ®йҷҚз»ҙпјҢж•°жҚ®йў„еӨ„зҗҶзӯүзӯүпјҢеҢ…еҗ«дәҶеёёи§Ғзҡ„еӨ§йғЁеҲҶжңәеҷЁеӯҰд№ ж–№жі•

ејәеӨ§ж•°жҚ®еә“В¶

1.йёўе°ҫиҠұж•°жҚ®йӣҶ:

from sklearn.datasets import load_iris
data = load_iris()
#еҜје…Ҙж•°жҚ®е’Ңж Үзӯҫ
data_X = data.data
data_y = data.target

2.жіўеЈ«йЎҝжҲҝд»·ж•°жҚ®йӣҶ:

from sklearn import datasets
loaded_data = datasets.load_boston()
#еҜје…Ҙж•°жҚ®
data_X = loaded_data.data
data_y = loaded_data.target

йҖҡз”ЁеӯҰд№ жЁЎејҸВ¶

йёўе°ҫиҠұж•°жҚ®йӣҶ:

from sklearn.model_selection import train_test_split
from sklearn import datasets

#kиҝ‘йӮ»еҮҪж•°
from sklearn.neighbors import KNeighborsClassifier
iris = datasets.load_iris()

#еҜје…Ҙж•°жҚ®е’Ңж Үзӯҫ
iris_X = iris.data
iris_y = iris.target

#еҲ’еҲҶдёәи®ӯз»ғйӣҶе’ҢжөӢиҜ•йӣҶж•°жҚ®
X_train, X_test, y_train, y_test = train_test_split(iris_X, iris_y, test_size=0.3)
#print(y_train)

#и®ҫзҪ®knnеҲҶзұ»еҷЁ
knn = KNeighborsClassifier()

#иҝӣиЎҢи®ӯз»ғ
knn.fit(X_train, y_train)

#дҪҝз”Ёи®ӯз»ғеҘҪзҡ„knnиҝӣиЎҢж•°жҚ®йў„жөӢ
print(knn.predict(X_test))
print(y_test)

жіўеЈ«йЎҝжҲҝд»·ж•°жҚ®йӣҶ:

import matplotlib.pyplot as plt
from sklearn import datasets
#и°ғз”ЁзәҝжҖ§еӣһеҪ’еҮҪж•°
from sklearn.linear_model import LinearRegression

#еҜје…Ҙж•°жҚ®йӣҶ
#иҝҷйҮҢе°Ҷе…ЁйғЁж•°жҚ®з”ЁдәҺи®ӯз»ғпјҢе№¶жІЎжңүеҜ№ж•°жҚ®иҝӣиЎҢеҲ’еҲҶ
#дёҠдҫӢдёӯ, е°Ҷж•°жҚ®еҲ’еҲҶдёәи®ӯз»ғе’ҢжөӢиҜ•ж•°жҚ®пјҢеҗҺйқўдјҡи®ІеҲ°дәӨеҸүйӘҢиҜҒ
loaded_data = datasets.load_boston()
data_X = loaded_data.data
data_y = loaded_data.target

#и®ҫзҪ®зәҝжҖ§еӣһеҪ’жЁЎеқ—
model = LinearRegression()
#и®ӯз»ғж•°жҚ®пјҢеҫ—еҮәеҸӮж•°
model.fit(data_X, data_y)

#еҲ©з”ЁжЁЎеһӢпјҢеҜ№ж–°ж•°жҚ®пјҢиҝӣиЎҢйў„жөӢпјҢдёҺеҺҹж ҮзӯҫиҝӣиЎҢжҜ”иҫғ
print(model.predict(data_X[:4,:]))
print(data_y[:4])
