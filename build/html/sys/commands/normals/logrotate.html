logrotateе‘Ҫд»ӨВ¶

зі»з»ҹз®ЎзҗҶе‘ҳеҸҜд»ҘдҪҝз”Ёlogrotate зЁӢеәҸз”ЁжқҘз®ЎзҗҶзі»з»ҹдёӯзҡ„жңҖж–°зҡ„дәӢд»¶гҖӮlogrotate иҝҳеҸҜд»Ҙз”ЁжқҘеӨҮд»Ҫж—Ҙеҝ—ж–Үд»¶

logrotate зҡ„й…ҚзҪ®ж–Үд»¶жҳҜ /etc/logrotate.conf:

compress йҖҡиҝҮgzip еҺӢзј©иҪ¬еӮЁд»ҘеҗҺзҡ„ж—Ҙеҝ—
nocompress дёҚйңҖиҰҒеҺӢзј©ж—¶пјҢз”ЁиҝҷдёӘеҸӮж•°
copytruncate з”ЁдәҺиҝҳеңЁжү“ејҖдёӯзҡ„ж—Ҙеҝ—ж–Үд»¶пјҢжҠҠеҪ“еүҚж—Ҙеҝ—еӨҮд»Ҫе№¶жҲӘж–ӯ
nocopytruncate еӨҮд»Ҫж—Ҙеҝ—ж–Үд»¶дҪҶжҳҜдёҚжҲӘж–ӯ
create mode owner group иҪ¬еӮЁж–Үд»¶пјҢдҪҝз”ЁжҢҮе®ҡзҡ„ж–Үд»¶жЁЎејҸеҲӣе»әж–°зҡ„ж—Ҙеҝ—ж–Үд»¶
nocreate дёҚе»әз«Ӣж–°зҡ„ж—Ҙеҝ—ж–Үд»¶
delaycompress е’Ң compress дёҖиө·дҪҝз”Ёж—¶пјҢиҪ¬еӮЁзҡ„ж—Ҙеҝ—ж–Үд»¶еҲ°дёӢдёҖж¬ЎиҪ¬еӮЁж—¶жүҚеҺӢзј©
nodelaycompress иҰҶзӣ– delaycompress йҖүйЎ№пјҢиҪ¬еӮЁеҗҢж—¶еҺӢзј©гҖӮ
errors address дё“еӮЁж—¶зҡ„й”ҷиҜҜдҝЎжҒҜеҸ‘йҖҒеҲ°жҢҮе®ҡзҡ„Email ең°еқҖ
ifempty еҚідҪҝжҳҜз©әж–Үд»¶д№ҹиҪ¬еӮЁпјҢиҝҷдёӘжҳҜ logrotate зҡ„зјәзңҒйҖүйЎ№гҖӮ
notifempty еҰӮжһңжҳҜз©әж–Үд»¶зҡ„иҜқпјҢдёҚиҪ¬еӮЁ
mail address жҠҠиҪ¬еӮЁзҡ„ж—Ҙеҝ—ж–Үд»¶еҸ‘йҖҒеҲ°жҢҮе®ҡзҡ„E-mail ең°еқҖ
nomail иҪ¬еӮЁж—¶дёҚеҸ‘йҖҒж—Ҙеҝ—ж–Үд»¶
olddir directory иҪ¬еӮЁеҗҺзҡ„ж—Ҙеҝ—ж–Үд»¶ж”ҫе…ҘжҢҮе®ҡзҡ„зӣ®еҪ•пјҢеҝ…йЎ»е’ҢеҪ“еүҚж—Ҙеҝ—ж–Үд»¶еңЁеҗҢдёҖдёӘж–Үд»¶зі»з»ҹ
noolddir иҪ¬еӮЁеҗҺзҡ„ж—Ҙеҝ—ж–Үд»¶е’ҢеҪ“еүҚж—Ҙеҝ—ж–Үд»¶ж”ҫеңЁеҗҢдёҖдёӘзӣ®еҪ•дёӢ
prerotate/endscript еңЁиҪ¬еӮЁд»ҘеүҚйңҖиҰҒжү§иЎҢзҡ„е‘Ҫд»ӨеҸҜд»Ҙж”ҫе…ҘиҝҷдёӘеҜ№пјҢиҝҷдёӨдёӘе…ій”®еӯ—еҝ…йЎ»еҚ•зӢ¬жҲҗиЎҢ
postrotate/endscript еңЁиҪ¬еӮЁд»ҘеҗҺйңҖиҰҒжү§иЎҢзҡ„е‘Ҫд»ӨеҸҜд»Ҙж”ҫе…ҘиҝҷдёӘеҜ№пјҢиҝҷдёӨдёӘе…ій”®еӯ—еҝ…йЎ»еҚ•зӢ¬жҲҗиЎҢ
daily жҢҮе®ҡиҪ¬еӮЁе‘ЁжңҹдёәжҜҸеӨ©
weekly жҢҮе®ҡиҪ¬еӮЁе‘ЁжңҹдёәжҜҸе‘Ё
monthly жҢҮе®ҡиҪ¬еӮЁе‘ЁжңҹдёәжҜҸжңҲ
rotate count жҢҮе®ҡж—Ҙеҝ—ж–Үд»¶еҲ йҷӨд№ӢеүҚиҪ¬еӮЁзҡ„ж¬Ўж•°пјҢ0 жҢҮжІЎжңүеӨҮд»ҪпјҢ5 жҢҮдҝқз•ҷ5 дёӘеӨҮд»Ҫ
tabootext [+] list и®©logrotate дёҚиҪ¬еӮЁжҢҮе®ҡжү©еұ•еҗҚзҡ„ж–Үд»¶пјҢзјәзңҒзҡ„жү©еұ•еҗҚжҳҜпјҡ.rpm-orig, .rpmsave, v, е’Ң ~
size size еҪ“ж—Ҙеҝ—ж–Үд»¶еҲ°иҫҫжҢҮе®ҡзҡ„еӨ§е°Ҹж—¶жүҚиҪ¬еӮЁпјҢSize еҸҜд»ҘжҢҮе®ҡ bytes (зјәзңҒ)д»ҘеҸҠKB (sizek)жҲ–иҖ…MB (sizem).

е®һдҫӢ(mysqlж—Ҙеҝ—еҲҮеүІ):

#vim mysql-log-rotate    е®ҡд№үж—Ҙеҝ—иҪ®ж»ҡзӯ–з•Ҙ
"/data/mysql_log/slow_log_3306.log" /data/mysql_log/error_log_3306.log {
     create 600 mysql mysql
     dateext
     notifempty
     daily
     maxage 60
     rotate 30
     missingok
     compress
     olddir /data/httplogs/old_log
 postrotate
     # just if mysqld is really running
     if test -x /usr/local/mysql/bin/mysqladmin && \
        /usr/local/mysql/bin/mysqladmin ping  -uroot -pPASSWORD -S /tmp/mysql_3306.sock &>/dev/null
     then
        /usr/local/mysql/bin/mysqladmin flush-logs -uroot -pPASSWORD -S /tmp/mysql_3306.sock
     fi
 endscript
 }
