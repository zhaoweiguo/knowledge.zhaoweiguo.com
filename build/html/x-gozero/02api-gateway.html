func NewServer(c RestConf, opts ...RunOption) (*Server, error) {
  // c.SetUp()еҒҡдёҖдәӣеҗҜеҠЁеүҚзҡ„еҮҶеӨҮе·ҘдҪң[жіЁ: иҝҷйғЁеҲҶеҸҜжҡӮеҝҪз•ҘпјҢд»ҘеҗҺдё“й—ЁеҶҚзңӢ]
  // зҺ°еңЁдё»иҰҒзҹҘйҒ“е®ғжҳҜеҒҡ:
  //    1. logxеҲқдҪҝеҢ–
  //    2. prometheus еҲқдҪҝеҢ–
  //    3. ејҖеҸ‘жЁЎејҸзӯүи®ҫзҪ®
  if err := c.SetUp(); err != nil {
    return nil, err
  }

  // ServerеҲқдҪҝеҢ– [жіЁ: еҫ…з»ҶеҢ–]
  server := &Server{
    ngin: newEngine(c),
    opts: runOptions{
      start: func(srv *engine) error {
        return srv.Start()
      },
    },
  }

  // йҖүйЎ№жЁЎејҸ: жү©еұ•зӣёе…іеҠҹиғҪ[жіЁ: иҝҷйғЁеҲҶеҸҜжҡӮеҝҪз•ҘпјҢд»ҘеҗҺдё“й—ЁеҶҚзңӢ]
  for _, opt := range opts {
    opt(server)
  }

  return server, nil
}

func newEngine(c RestConf) *engine {
  // жҠҠ RestConf зұ»еһӢж•°жҚ®е°ҒиЈ…еҲ° engine зұ»еһӢдёӯ
  // жң¬иҙЁ: еҲқдҪҝеҢ– engine зҡ„"conf"еӯ—ж®ө
  srv := &engine{
    conf: c,
  }
  // иҝҷеқ—дё»иҰҒжҳҜ"иҙҹиҚ·"зӣёе…іи®ҫзҪ® [жіЁ: иҝҷйғЁеҲҶеҸҜжҡӮеҝҪз•ҘпјҢд»ҘеҗҺдё“й—ЁеҶҚзңӢ]
  // жң¬иҙЁ: еҲқдҪҝеҢ– engine зҡ„"shedder"е’Ң"priorityShedder"еӯ—ж®ө
  if c.CpuThreshold > 0 {
    srv.shedder = load.NewAdaptiveShedder(load.WithCpuThreshold(c.CpuThreshold))
    srv.priorityShedder = load.NewAdaptiveShedder(load.WithCpuThreshold(
      (c.CpuThreshold + topCpuUsage) >> 1))
  }
  return srv
}

func StartHttp(host string, port int, handler http.Handler) error {
  return start(host, port, handler,
      func(srv *http.Server) error {
        return srv.ListenAndServe()
      }
  )
}

func start(host string, port int, handler http.Handler, run func(srv *http.Server) error) (err error) {
  server := &http.Server{
    Addr:    fmt.Sprintf("%s:%d", host, port),
    Handler: handler,
  }
  return run(server)
}

func (s *engine) bindRoute(fr featuredRoutes, router httpx.Router, metrics *stat.Metrics,
  route Route, verifier func(chain alice.Chain) alice.Chain) error {
  chain := alice.New(
    handler.TracingHandler,
    s.getLogHandler(),
    handler.PrometheusHandler(route.Path),
    handler.MaxConns(s.conf.MaxConns),
    handler.BreakerHandler(route.Method, route.Path, metrics),
    handler.SheddingHandler(s.getShedder(fr.priority), metrics),
    handler.TimeoutHandler(time.Duration(s.conf.Timeout)*time.Millisecond),
    handler.RecoverHandler,
    handler.MetricHandler(metrics),
    handler.MaxBytesHandler(s.conf.MaxBytes),
    handler.GunzipHandler,
  )
  chain = s.appendAuthHandler(fr, chain, verifier)

  for _, middleware := range s.middlewares {
    chain = chain.Append(convertMiddleware(middleware))
  }
  handle := chain.ThenFunc(route.Handler)

  return router.Handle(route.Method, route.Path, handle)
}
