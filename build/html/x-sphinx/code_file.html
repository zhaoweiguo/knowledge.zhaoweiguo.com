10          $INSECURE_REGISTRY

MIME types: text/x-java-properties

4clientPort=2181

<VirtualHost *:80>
        ServerName a.com
        ServerAdmin webmaster@localhost
        DocumentRoot /home/data/www/a
        ErrorLog /home/data/log/a_error.log
        CustomLog /home/data/log/a_access.log combined
</Virtualhost>
<VirtualHost *:80>
        ServerName b.com
    ServerAdmin webmaster@localhost

</Virtualhost>
